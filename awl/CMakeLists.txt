##
## Expand Qt macros in source files
##

QT4_WRAP_CPP(awl_mocs
    floatentry.h
    pitchedit.h
    pitchlabel.h
    posedit.h
    sigedit.h
    tempoedit.h
    tempolabel.h
)

##
## List of source files to compile
##

file(GLOB awl_source_files
    awl.cpp
    floatentry.cpp
    pitchedit.cpp
    pitchlabel.cpp
    posedit.cpp
    sigedit.cpp
    tempoedit.cpp
    tempolabel.cpp
    utils.cpp
)

##
## Define target
##

add_library(awl STATIC
    ${awl_source_files}
    ${awl_mocs}
)

##
## Append to the list of translations
##

set(FILES_TO_TRANSLATE
    ${FILES_TO_TRANSLATE}
    ${awl_source_files}
    CACHE INTERNAL ""
)

##
## Compilation flags and target name
##

set_target_properties(awl
    PROPERTIES OUTPUT_NAME los_awl
)
